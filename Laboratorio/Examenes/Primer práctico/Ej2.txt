import numpy as np
import matplotlib.pyplot as plt

def busquedaIncremental(f,a,b,n):
    x = np.linspace(a,b,n+1)
    intervalos = np.zeros((n,2))
    contador = 0
    for i in range(n):
        if f(x[i])*f(x[i+1]) < 0:
            intervalos[contador] = x[i:i+2]
            contador +=1
    intervalos = intervalos[:contador]
    return intervalos
